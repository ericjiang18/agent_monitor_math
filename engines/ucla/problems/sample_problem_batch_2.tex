\begin{problem*}
  For $n \geq 1$, consider the double cover $\pi\from S^n \to \RP^n$. Is the image of the map
  \[
  \pi^*\from \KO^{-1}(\RP^n) \to \KO^{-1}(S^n)
  \]
  $2$-divisible?
\end{problem*}
