\begin{problem*}
  For $n \geq 1$, consider the double cover $\pi\from S^n \to \RP^n$. Is the image of the map in the reduced $\KO$-theory
  \[
  \pi^*\from \tilde{\KO}^{-1}(\RP^n) \to \tilde{\KO}^{-1}(S^n)
  \]
  $2$-divisible?
\end{problem*}
